  • {{300000 | currency: '$' : 4}} 前边加$符，保留小数点后四位小数（默认保留小数点后两位小数）
  • {{300000 | number: '3'}} 保留小数点后3位小数

  • 大小写转换

  • {{'lower cap string' | uppercase }} 转换为大写
  • {{'LOWER CAP STRING' | lowercase }} 转换为小写

  • 数字的截取

  • {{123456789 | limitTo:5}} 从前边筛选出5个数字
  • {{123456789 | limitTo:-5}} 从后边筛选出5个数字

  • 内容的筛选(过滤器根据表达式过滤不包含过滤器中的内容)

  • {{ [{"age": 20,"id": 10,"name": "iphone"}, {"age": 12,"id": 11,"name": "sunm xing"}, {"age": 44,"id": 12,"name": "test abc"} ] | filter:{'name':'iphone'} }} 查找name为iphone的行
  • {{ [{"age": 20,"id": 10,"name": "iphone"}, {"age": 12,"id": 11,"name": "sunm xing"}, {"age": 44,"id": 12,"name": "test abc"} ] | filter:{'name':'i'} }} 查找name中有i的行
  • {{ 'asdfg' | aa }} 自定义过滤器
  • {{ [{"age": 20,"id": 10,"name": "iphone"}, {"age": 12,"id": 11,"name": "sunm xing"}, {"age": 44,"id": 12,"name": "test abc"} ] | filter }} 不进行筛选（按原本的内容显示）

  • 内容的顺序排列( 默认正序asc,倒序添加-负号)

  • {{ [{"age": 20,"id": 10,"name": "iphone"}, {"age": 12,"id": 11,"name": "sunm xing"}, {"age": 44,"id": 12,"name": "test abc"} ] | orderBy: "asc" }} 正序显示（asc 可以省略 因为默认值为asc）
  • {{ [{"age": 20,"id": 10,"name": "iphone"}, {"age": 12,"id": 11,"name": "sunm xing"}, {"age": 44,"id": 12,"name": "test abc"} ] | orderBy: "-" }} 倒序显示
  • {{ [{"age": 20,"id": 10,"name": "iphone"}, {"age": 12,"id": 11,"name": "sunm xing"}, {"age": 44,"id": 12,"name": "test abc"} ] | orderBy: "age" }} 按age的升序显示
  • {{ [{"age": 20,"id": 10,"name": "iphone"}, {"age": 12,"id": 11,"name": "sunm xing"}, {"age": 44,"id": 12,"name": "test abc"} ] | orderBy: "-age" }} 按age的降序显示
